To get started with AegisTrader:

1. Check the deployment status:
   kubectl get all -n {{ .Release.Namespace }}

2. Wait for all pods to be ready:
   kubectl wait --for=condition=ready pod -l app.kubernetes.io/instance={{ .Release.Name }} -n {{ .Release.Namespace }} --timeout=300s

3. Port forward to access services locally:
   {{- if index .Values "monitor-api" "enabled" }}
   # Monitor API
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ .Release.Name }}-monitor-api {{ index .Values "monitor-api" "service" "port" }}:{{ index .Values "monitor-api" "service" "port" }}
   {{- end }}

   {{- if index .Values "monitor-ui" "enabled" }}
   # Monitor UI
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ .Release.Name }}-monitor-ui {{ index .Values "monitor-ui" "service" "port" }}:{{ index .Values "monitor-ui" "service" "port" }}
   {{- end }}

   # NATS
   kubectl port-forward -n {{ .Release.Namespace }} svc/{{ .Release.Name }}-nats 4222:4222

4. Access the services:
   {{- if index .Values "monitor-ui" "enabled" }}
   Monitor UI: http://localhost:{{ index .Values "monitor-ui" "service" "port" }}
   {{- end }}
   {{- if index .Values "monitor-api" "enabled" }}
   Monitor API: http://localhost:{{ index .Values "monitor-api" "service" "port" }}/docs
   {{- end }}

5. Check service logs:
   kubectl logs -f -n {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }} --all-containers=true --prefix=true
